# 매출 패턴 꺾은선 차트 데이터 타입 관련 논의

> 참여자: 이영재, 정한울

```tsx
interface datum {
  amount: number | string;
  unit: string;
}

interface data {
  mainX: datum[]; // 그래프 x축 기준: 시간, 기간
  subX: datum[]; // 그래프 x축에는 표시되지 않지만 툴팁에는 보이는 x
  mainY: datum[];
  subY: datum[];
}

/**
 * @description 꺾은선 차트 컴포넌트 (자세한 사용법은 스토리북 문서 참고)
 */
interface LineChartProps {
  viewBoxWidth: number;
  viewBoxHeight: number;
  hasXAxis?: boolean;
  hasGradient?: boolean;
  showXGuideLine?: boolean;
  showYGuideLine?: boolean;
  primarySeries: series;
  secondarySeries?: series;
  yGuideLineCount: number;
  activeTooltip?: boolean;
  tooltipContent?: (...args: string[]) => string;
  chartTitle?: string;
  chartDescription?: string;
}
```

프론트에서 차트 그려줄 때 data = [실시간, 월요일 평균]

이거는 매출 패턴 꺾은선 차트 관련 data 데이터 타입 관련 논의야
